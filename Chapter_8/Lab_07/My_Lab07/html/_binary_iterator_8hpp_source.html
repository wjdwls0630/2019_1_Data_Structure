216 template <typename T>
217 void BinaryIterator<T>::CastListRecur(BNodeType<T>* Node){
218 // left -> cur -> right
219  if (Node->IsLeaf()) { // if node is leaf , print cur node only(Base Case)
220  this->m_List[this->length] = Node;
221  this->length++;
222  return ;
223  } else { // if node has child
224  // left -> cur -> right
225  if (Node->GetLeftLink()!= nullptr) {
226  this->CastListRecur(Node->GetLeftLink());
227  }
228  this->m_List[this->length] = Node;
229  this->length++;
230  if (Node->GetRightLink()!= nullptr) {
231  this->CastListRecur(Node->GetRightLink());
232  }
233  }
234 }
