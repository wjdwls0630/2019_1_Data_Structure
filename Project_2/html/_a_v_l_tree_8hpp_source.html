179 // Get a BalanceFactor of Node.
180 template<typename T>
181 int AVLTree<T>::GetBalanceFactor(BNodeType<T> *Node) {
182  int H_Left;
183  int H_Right;
184  if (Node == nullptr) {
185  return 0;
186  }
187  if (Node->IsLeaf()) {
188  //no difference height
189  return 0;
190  } else if (Node->GetLeftLink()==nullptr||Node->GetRightLink()==nullptr) {
191  //if empty node has, calculate it -1
192  if (Node->GetLeftLink()!=nullptr) {
193  //has only left child
194  H_Left = this->HeightNode(Node->GetLeftLink());
195  return H_Left + 1;
196  } else {
197  //has only right child
198  H_Right = this->HeightNode(Node->GetRightLink());
199  return -1 - H_Right;
200  }
201  } else {
202  //if node has both child
203  H_Left = this->HeightNode(Node->GetLeftLink());
204  H_Right = this->HeightNode(Node->GetRightLink());
205  return H_Left - H_Right;
206  }
207 }

312 template<typename T>
313 BNodeType<T> *AVLTree<T>::ReBalance(BNodeType<T> *Parent_Node) {
314  int bf = this->GetBalanceFactor(Parent_Node); //balance factor
315  //if bf is 0 parent node is 0, there is no case that rise error that Parent Node has no link
316  if (bf>1) {
317  //case 1: unbalance in left
318  if (this->GetBalanceFactor(Parent_Node->GetLeftLink()) > 0) {
319  //problem in left child
320  Parent_Node = this->RotationLL(Parent_Node);
321  } else {
322  //problem in right child
323  Parent_Node = this->RotationLR(Parent_Node);
324  }
325  } else if (bf < -1) {
326  //case 2: unbalance in right
327  if (this->GetBalanceFactor(Parent_Node->GetRightLink()) < 0) {
328  //problem in right child
329  Parent_Node = this->RotationRR(Parent_Node);
330  } else {
331  //problem in left child
332  Parent_Node = this->RotationRL(Parent_Node);
333  }
334  }
335  return Parent_Node;

360  //overlap data is already filtered in Add func
361  if (this->bt_Comparer.Compare(data, Parent_Node->GetData())) {
362  //ASCENDING(1) data < Node->GetData()
363  //DESCENDING(0) data > Node->GetData()
364  if (Parent_Node->GetLeftLink()!= nullptr) {
365  //if node has left child
366  Parent_Node->SetLeftLink(this->AddAVLRecur(Parent_Node->GetLeftLink(), data));
367  } else {
368  //if node has no left child
369  BNodeType<T> * NewNode = new BNodeType<T>;
370  NewNode->SetData(data);
371  Parent_Node->SetLeftLink(NewNode);
372  }
373  } else if (this->bt_Comparer.Op_Compare(data, Parent_Node->GetData())){
374  //ASCENDING(1) data > Node->GetData()
375  //DESCENDING(0) data < Node->GetData()
376  if (Parent_Node->GetRightLink()!= nullptr) {
377  //if node has left child
378  Parent_Node->SetRightLink(this->AddAVLRecur(Parent_Node->GetRightLink(), data));
